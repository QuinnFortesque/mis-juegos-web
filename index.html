               });

           } else {

               autocompleteResults.classList.add('hidden');

           }

       });

       // Ocultar autocomplete al hacer click fuera

       document.addEventListener('click', (e) => {

           if (!titleInput.contains(e.target) && !autocompleteResults.contains(e.target)) {

               autocompleteResults.classList.add('hidden');

           }

       });

       // Actualizar slider display

       ratingInput.addEventListener('input', (e) => {

           ratingDisplay.innerText = e.target.value;

           // Cambiar color según puntuación

           const val = parseInt(e.target.value);

           ratingDisplay.className = `text-2xl font-black ${val >= 90 ? 'text-yellow-400' : val >= 70 ? 'text-green-400' : val >= 50 ? 'text-cyan-400' : 'text-red-400'}`;

       });

       // Cargar juegos desde Firestore en tiempo real

       function loadGames() {

           if (!userId) return;

           const gamesRef = collection(db, 'artifacts', appId, 'users', userId, 'games');

         

           onSnapshot(gamesRef, (snapshot) => {

               gamesData = [];

               snapshot.forEach(doc => {

                   gamesData.push({ id: doc.id, ...doc.data() });

               });

               // Ordenar por fecha de creación descendente por defecto

               gamesData.sort((a, b) => b.createdAt - a.createdAt);

               renderCurrentTab();

           }, (error) => {

               console.error("Error escuchando juegos:", error);

           });

       }

       // Guardar juego

       form.addEventListener('submit', async (e) => {

           e.preventDefault();

           if (!userId) return;

           const title = titleInput.value.trim();

           let cover = coverInput.value.trim();

         

           // Generar placeholder si no hay cover

           if (!cover) {

               cover = `https://placehold.co/600x800/1e293b/06b6d4?text=${encodeURIComponent(title)}`;

           }

           const newGame = {

               title,

               cover,

               platform: document.getElementById('platform').value,

               status: document.querySelector('input[name="status"]:checked').value,

               rating: parseInt(ratingInput.value),

               notes: document.getElementById('notes').value.trim(),

               isGoty: false,

               createdAt: Date.now()

           };

           const submitBtn = form.querySelector('button[type="submit"]');

           const originalText = submitBtn.innerHTML;

           submitBtn.innerHTML = '<i class="ph ph-spinner animate-spin"></i> Guardando...';

           submitBtn.disabled = true;

           try {

               await addDoc(collection(db, 'artifacts', appId, 'users', userId, 'games'), newGame);

               form.reset();

               ratingDisplay.innerText = '85';

               ratingDisplay.className = 'text-2xl font-black text-cyan-400';

               coverInput.value = '';

           } catch (error) {

               console.error("Error al guardar:", error);

           } finally {

               submitBtn.innerHTML = originalText;

               submitBtn.disabled = false;

           }

       });

       // Eliminar juego

       window.deleteGame = async (id) => {

           if (!userId) return;

           try {

               await deleteDoc(doc(db, 'artifacts', appId, 'users', userId, 'games', id));

           } catch (error) {

               console.error("Error al eliminar:", error);

           }

       };

       // Marcar/Desmarcar GOTY

       window.toggleGoty = async (id, currentGotyStatus) => {

           if (!userId) return;

         

           try {

               // Si lo estamos marcando como GOTY, primero quitamos el GOTY actual

               if (!currentGotyStatus) {

                   const currentGoty = gamesData.find(g => g.isGoty);

                   if (currentGoty) {

                       await updateDoc(doc(db, 'artifacts', appId, 'users', userId, 'games', currentGoty.id), { isGoty: false });

                   }

               }

               // Actualizamos el juego seleccionado

               await updateDoc(doc(db, 'artifacts', appId, 'users', userId, 'games', id), { isGoty: !currentGotyStatus });

           } catch (error) {

               console.error("Error al actualizar GOTY:", error);

           }

       };

     

       // Utilidad para obtener el color del estado

       function getStatusColor(status) {

           switch(status) {

               case 'Jugando': return 'bg-blue-500/20 text-blue-400 border-blue-500/30';

               case 'Pendiente': return 'bg-purple-500/20 text-purple-400 border-purple-500/30';

               case 'Completado': return 'bg-green-500/20 text-green-400 border-green-500/30';

               case 'Dropeado': return 'bg-red-500/20 text-red-400 border-red-500/30';

               default: return 'bg-slate-700 text-slate-300 border-slate-600';

           }

       }

       window.switchTab = (tabId) => {

           currentTab = tabId;

           const btnAll = document.getElementById('tab-all');

           const btnTop = document.getElementById('tab-top');

           const viewAll = document.getElementById('view-all');

           const viewTop = document.getElementById('view-top');

           if (tabId === 'all') {

               btnAll.className = "tab-btn px-6 py-2.5 rounded-lg font-medium text-sm transition-all bg-slate-700 text-white shadow";

               btnTop.className = "tab-btn px-6 py-2.5 rounded-lg font-medium text-sm transition-all text-slate-400 hover:text-slate-200";

               viewAll.classList.remove('hidden');

               viewTop.classList.add('hidden');

           } else {

               btnTop.className = "tab-btn px-6 py-2.5 rounded-lg font-medium text-sm transition-all bg-slate-700 text-white shadow";

               btnAll.className = "tab-btn px-6 py-2.5 rounded-lg font-medium text-sm transition-all text-slate-400 hover:text-slate-200";

               viewTop.classList.remove('hidden');

               viewAll.classList.add('hidden');

           }

           renderCurrentTab();

       };

       // Renderizar Top 10

       function renderTop10() {

           topGrid.innerHTML = '';

         

           // Filtrar juegos que tienen nota (opcional, por si acaso) y ordenar por nota descendente

           const sortedGames = [...gamesData].sort((a, b) => b.rating - a.rating).slice(0, 10);

           if (sortedGames.length === 0) {

               topGrid.innerHTML = `

                   <div class="flex flex-col items-center justify-center p-12 text-slate-500 bg-slate-800/50 border border-dashed border-slate-700 rounded-2xl">

                       <i class="ph-fill ph-medal text-6xl mb-4 opacity-50"></i>

                       <p class="text-lg">No hay juegos para rankear.</p>

                   </div>

               `;

               return;

           }

           sortedGames.forEach((game, index) => {

               const row = document.createElement('div');

               const isFirst = index === 0;

             

               // Estilos especiales para el podio

               let rankStyle = "bg-slate-700 text-slate-300";

               if(index === 0) rankStyle = "bg-yellow-500 text-yellow-900 shadow-[0_0_15px_rgba(234,179,8,0.5)]";

               if(index === 1) rankStyle = "bg-slate-300 text-slate-800";

               if(index === 2) rankStyle = "bg-amber-700 text-amber-100";

               row.className = `flex items-center gap-4 bg-slate-800 p-4 rounded-xl border border-slate-700 ${isFirst ? 'transform scale-[1.02] border-yellow-500/50' : ''} hover:bg-slate-750 transition-colors`;

             

               row.innerHTML = `

                   <div class="flex-shrink-0 w-12 h-12 flex items-center justify-center rounded-lg font-black text-xl ${rankStyle}">

                       #${index + 1}

                       <div class="flex items-center gap-2 text-xs mt-1">

                           <span class="text-slate-400">${game.platform}</span>

                           <span class="w-1 h-1 rounded-full bg-slate-600"></span>

                           <span class="${game.status === 'Completado' ? 'text-green-400' : 'text-slate-400'}">${game.status}</span>

                           ${game.isGoty ? '<span class="w-1 h-1 rounded-full bg-slate-600"></span><span class="text-yellow-400 font-bold"><i class="ph-fill ph-crown"></i> GOTY</span>' : ''}
